1. 프로젝트 개요

DAACS란?

DAACS = Dynamic Agentic AI Coding System

사용자의 자연어 요구사항을 받아 완전한 풀스택 애플리케이션을 자동으로 설계, 구현, 검증하는 지능형 멀티 에이전트 시스템

핵심 특징

🤝
멀티 에이전트 협업
Orchestrator, Backend, Frontend, Verifier가 협력
🔧
Agentic CLI
Codex, Claude, Gemini로 실제 파일 생성
🔄
자가 치유
에러 시 자동 분석 및 수정 Feedback Loop
⚡
병렬 실행
Backend/Frontend 동시 개발

1.1 프로젝트 배경 및 필요성

class LLMSource(ABC):
    """LLM 소스 베이스 클래스 (추상)"""
    @abstractmethod
    def invoke(self, prompt: str) -> str: pass
    @abstractmethod
    def invoke_structured(self, prompt: str) -> Dict: pass

class CLIAssistantLLMSource(LLMSource):
    """CLI Assistant의 내장 LLM 사용 (Claude Code → Claude LLM, Codex → GPT LLM)"""
    def invoke(self, prompt: str) -> str:
        result = self.client.execute(prompt)  # cli_executor.py 호출
        if result.startswith("Error:"):
            if self.fallback_config:
                # Fallback: Plugin LLM으로 전환
                fallback_source = PluginLLMSource(...)
                return fallback_source.invoke(prompt)

class PluginLLMSource(LLMSource):
    """플러그인 LLM 사용 (OpenAI, Anthropic, Google API 직접 호출)"""
    def invoke(self, prompt: str) -> str:
        if self.provider == "openai":
            return self.llm.chat.completions.create(...).choices[0].message.content
        elif self.provider == "anthropic":
            return self.llm.messages.create(...).content[0].text

def execute(self, prompt: str) -> str:
    """CLI를 비대화형 모드(exec)로 실행하여 결과를 받아옵니다."""
    
    if self.cli_type == "claude_code":
        # Claude CLI: --dangerously-skip-permissions로 자동 승인
        cmd = ["claude.cmd", "--dangerously-skip-permissions"]
        input_str = prompt  # 프롬프트는 stdin으로 전달
    
    elif self.cli_type == "gemini":
        # Gemini CLI: -s (silent mode)로 확인 프롬프트 스킵
        cmd = ["gemini.cmd", "-s"]
        input_str = prompt
    
    else:  # Codex (기본값)
        # Codex CLI: exec 모드 + sandbox 권한 부여
        permissions = f'sandbox_permissions={json.dumps(["disk-full-access"])}'
        cmd = ["npx", "@openai/codex", "exec", "--sandbox", "danger-full-access", "-c", permissions]
        input_str = prompt
    
    # subprocess.run으로 실행
    result = subprocess.run(cmd, input=input_str, cwd=self.cwd, ...)

Rate Limiter (Gemini 무료 티어용)

class GeminiRateLimiter:
    """Gemini 무료 티어 (분당 2회) 제한을 위한 Rate Limiter"""
    _last_request_time = 0
    _min_interval = 30.0  # 30초 대기 (여유 있게 설정)

    @classmethod
    def wait_for_slot(cls):
        elapsed = time.time() - cls._last_request_time
        if elapsed < cls._min_interval:
            wait_time = cls._min_interval - elapsed
            time.sleep(wait_time)  # 다음 슬롯까지 대기
        cls._last_request_time = time.time()

6.1 조건부 분기 로직 (실제 코드)

Planning 후 분기 (라인 87-104)

def decide_after_planning(state: DAACSState) -> str:
    """Backend/Frontend 필요 여부에 따라 분기"""
    needs_backend = state.get("needs_backend", False)   # Orchestrator가 설정한 값
    needs_frontend = state.get("needs_frontend", False)

    if needs_backend or needs_frontend:
        return "parallel_execution"  # → start_parallel → Backend/Frontend 병렬 실행
    else:
        return "deliver"             # → 바로 완료 (코드 생성 불필요)

Judgment 후 분기 (라인 121-144)

def decide_after_judgment(state: DAACSState) -> str:
    """재작업 필요 여부 및 iteration 상한 확인"""
    needs_rework = state.get("needs_rework", False)
    iteration_count = state.get("iteration_count", 0)
    max_iterations = state.get("max_iterations", 10)

    # 1. 최대 반복 횟수 도달 → 강제 완료
    if iteration_count >= max_iterations:
        return "save_context"

    # 2. 재작업 필요 → Replanning으로
    if needs_rework:
        return "replan"

    # 3. 정상 완료
    return "save_context"
⚠️ 무한 루프 방지: max_iterations(10)과 max_failures(5) 두 가지 상한으로 강제 종료
